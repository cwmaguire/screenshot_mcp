import os

app = FastAPI()

class ScreenshotRequest(BaseModel):

@app.post("/screenshot")
async def receive_screenshot(request: ScreenshotRequest):
    try:
        # Decode base64 image
        image_bytes = base64.b64decode(request.image_data)
        image = Image.open(io.BytesIO(image_bytes))
        
        # Save the image
        save_path = os.getenv('SAVE_PATH', 'screenshots')
        os.makedirs(save_path, exist_ok=True)
        filename = f"screenshot_{len(os.listdir(save_path)) + 1}.png"
        image.save(os.path.join(save_path, filename))
        
        return {"message": "Screenshot saved successfully", "filename": filename}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
